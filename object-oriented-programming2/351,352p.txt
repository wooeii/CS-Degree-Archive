15. Rect 객체 두 개를 받아서 같은 크기인지를 체크하는 일반함수 is_equal()을 작성하여 보자. 다음과 같은 함수 원형을 가정한다.

bool is_equal(Rect r1, Rect r2);


#include <iostream>
#include <string>
using namespace std;

class Rect {
	int width, height;
public:
	Rect(int width=0, int height=0): width(width), height(height) {}
	~Rect(){}
	void setvalue(int a, int b) { this->width = a, this->height = b; }
	int getArea() const { return width * height; }
	int getPerimeter() const { return (width + height) * 2; }
	void print();
};
void Rect::print()
{
	cout << "면적은" << getArea() << ", 둘레는" << getPerimeter() << "입니다." << endl;
}
bool is_equal(const Rect r1, const Rect r2)
{
	return r1.getArea() == r2.getArea() && r1.getPerimeter() == r2.getPerimeter();
}

int main()
{
	Rect* const r1= new Rect();
	Rect* const r2 = new Rect();
	r1->setvalue(5, 6);
	r2->setvalue(5, 6);

	if (is_equal(*r1, *r2)) {
		cout << "두 사각형은 같습니다." << endl;
	}
	else {
		cout << "두 사각형은 다릅니다." << endl;
	}
	delete r1;
	delete r2;
	return 0;
}
